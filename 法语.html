        function getCategoryColor(cat) {

            const colors = {

                grammar: 'bg-purple-100 text-purple-700',

                vocabulary: 'bg-blue-100 text-blue-700',

                culture: 'bg-orange-100 text-orange-700',

                expression: 'bg-green-100 text-green-700'

            };

            return colors[cat] || 'bg-gray-100 text-gray-700';

        }


        function getCategoryName(cat) {

            const names = {

                grammar: currentLanguage === 'zh' ? '语法' : currentLanguage === 'en' ? 'Grammar' : 'Grammaire',

                vocabulary: currentLanguage === 'zh' ? '词汇' : currentLanguage === 'en' ? 'Vocabulary' : 'Vocabulaire',

                culture: currentLanguage === 'zh' ? '文化' : currentLanguage === 'en' ? 'Culture' : 'Culture',

                expression: currentLanguage === 'zh' ? '表达' : currentLanguage === 'en' ? 'Expression' : 'Expression'

            };

            return names[cat] || cat;

        }


        function formatDate(dateString) {

            const date = new Date(dateString);

            return date.toLocaleDateString(currentLanguage === 'zh' ? 'zh-CN' : currentLanguage === 'en' ? 'en-US' : 'fr-FR');

        function nextQuestion() {

            // Select random note

            const note = notes[Math.floor(Math.random() * notes.length)];

            const questionType = Math.random() > 0.5 ? 'translation' : 'context';

            

            let question, answer, options;

            

            if (questionType === 'translation') {

                question = currentLanguage === 'zh' ? `"${note.french}" 是什么意思？` : 

                          currentLanguage === 'en' ? `What does "${note.french}" mean?` :

                          `Que signifie "${note.french}" ?`;

                answer = note.translation;

                

                // Generate wrong options

                const wrongOptions = notes

                    .filter(n => n.id !== note.id)

                    .sort(() => 0.5 - Math.random())

                    .slice(0, 3)

                    .map(n => n.translation);

                

                options = [answer, ...wrongOptions].sort(() => 0.5 - Math.random());

            } else {

                question = currentLanguage === 'zh' ? `以下哪句法语表达了："${note.translation}"？` :

                          currentLanguage === 'en' ? `Which French phrase means: "${note.translation}"?` :

                          `Quelle phrase française signifie : "${note.translation}" ?`;

                answer = note.french;

                

                const wrongOptions = notes

                    .filter(n => n.id !== note.id)

                    .sort(() => 0.5 - Math.random())

                    .slice(0, 3)

                    .map(n => n.french);

                

                options = [answer, ...wrongOptions].sort(() => 0.5 - Math.random());

            }

            

            currentQuiz = { answer, note };

            

            document.getElementById('quiz-question').textContent = question;

            document.getElementById('quiz-hint').textContent = currentLanguage === 'zh' ? '选择正确答案' : 'Select the correct answer';

            

            const optionsContainer = document.getElementById('quiz-options');

            optionsContainer.innerHTML = options.map((opt, idx) => `

                <button onclick="selectQuizOption('${opt.replace(/'/g, "\\'")}', this)" 

                        class="w-full text-left p-4 rounded-xl border-2 border-gray-200 hover:border-french-blue hover:bg-blue-50 transition-all text-lg font-medium">

                    ${String.fromCharCode(65 + idx)}. ${opt}

                </button>

            `).join('');

            

            optionsContainer.classList.remove('hidden');

            document.getElementById('quiz-result').classList.add('hidden');

        }


        function selectQuizOption(selected, btnElement) {

                    btn.classList.add('bg-red-100', 'border-red-500', 'text-red-800');

        function importData(input) {

            const file = input.files[0];

            if (!file) return;

            

            const reader = new FileReader();

            reader.onload = (e) => {

                try {

                    const imported = JSON.parse(e.target.result);

                    if (Array.isArray(imported)) {

                        notes = [...imported, ...notes];

                        saveToStorage();

                        renderNotes();

                        updateStats();

                        alert(currentLanguage === 'zh' ? '导入成功！' : 'Import successful!');

                    }

                } catch (err) {

                    alert(currentLanguage === 'zh' ? '文件格式错误' : 'Invalid file format');

                }

            };

            reader.readAsText(file);

            input.value = '';

        }
